    # 回忆 promise 的调用机制

    1. 它是一个构造函数，接收一个执行器函数为参数，这个执行器函数（executor）在创建 promise 实例的时候就会被立即调用执行（即执行器函数的代码是同步执行的）。
    2. executor 接收两个函数参数：resolve、reject。resolve 函数代表执行成功的回调，可以传入一个参数代表成功结果，reject 函数代表执行失败的回调，可以传入一个参数代表失败原因。
    3. 一个 promsie 实例有三种状态：pending、fulfilled（成功）、rejected（失败），且该状态一经修改就不可变了。执行 resolve 函数将 promsie 实例的状态更改为 fulfilled，执行 reject 函数将 promise 实例的状态更改为 rejected。
    4. promise 实例上有个 then 方法，该方法接收两个回调函数，并返回一个 promise 实例，所以该方法支持链式调用。then 方法的第一个回调表示成功回调，当 executor 内的代码调用 resolve 函数时就会执行该回调，当 executor 内调用 reject 函数时则会调用 then 的第二个回调函数，表示失败的回调。
    5. then 方法返回的 promise 的状态取决于 then 传递的两个回调函数执行之后是否报错，而不论执行的是成功回调还是失败回调；还有若 then 方法返回的是 promise 实例，则取决于这个 promise 实例的状态。
    6. then 链实现的原理：需要在实例上保存两个队列：分别是成功回调的队列和失败回调的队列。因为我们是为了处理异步的情况，所以 resolve 和 reject 我们其实不知道他到底是异步执行还是同步执行的。所以我们需要在 then 中判断当前实例的状态，如果状态还是 pending 态，则需要将回调存储，等到状态变更后进行调用（在 resolve 和 reject 中进行判断）。
    7. 我们还需要对 then 中传入的回调的返回值做判断：分为 promise 实例和非 promise 实例两种情况。是 promise 实例的情况又要判断它是否等于调用这个 then 的实例，避免自调用的情况（实现是否是自调用需要使用到异步--setTimeout，因为同步执行获取不到当前 then 返回的 promise）。如果不是自调用则直接调用这个 promise 实例的 then 方法，传入 reslolve，reject 即可。如果是非 promise 实例则直接 resolve（返回值即可）。
    8. catch 方法的实现：其实就是 then 不传 成功回调即可。
    9. then 链还可以不传参数，所以要对没传参数或者传的参数不是函数的情况做处理，给他一个默认的函数。
    10. all 方法的实现：是一个静态方法，主要是要保证输出的结果的顺序和传入的数组的顺序一样，需要一个记录下标来实现。

    # 具体实现

    const PENDING = "PENDING"; // 等待
const FULFILLED = "FULFILLED"; // 成功
const REJECTED = "REJECTED"; // 失败
class MyPromise {
  status = PENDING; // 初始化实例状态为 pending
  value = undefined; // resolve 被调用时传递的代表成功的值
  reason = undefined; // reject 被调用时传递的代表失败的原因
  successCallback = [];
  failCallback = [];
  constructor(executor) {
    try {
      // 实例化 promise 时，执行 executor，并传入 resolve 和 reject
      executor(this.resolve, this.reject);
    } catch (e) {
      this.reject(e);
    }
  }
  // 使用箭头函数的原因是让 this 指向为 promise 实例
  resolve = (value) => {
    if (this.status !== PENDING) return; // 状态一旦已经发生改变，则不再允许被更改
    this.status = FULFILLED; // 执行 resolve 将promise实例状态改为成功
    this.value = value;
    // this.successCallback && this.successCallback(this.value); // 异步时等待状态变化执行 then.else
    while (this.successCallback.length) {
      this.successCallback.shift()(); // 每执行一次则长度减一，直到长度为0
    }
  };
  reject = (reason) => {
    if (this.status !== PENDING) return;
    this.status = REJECTED; // 执行 reject 将promise实例状态改为失败
    this.reason = reason;
    // this.failCallback && this.failCallback(this.reason); // 异步时等待状态变化执行 then.else
    while (this.failCallback.length) this.failCallback.shift()();
  };
  // 添加的默认值是在处理then未传参数的情况或者传的参数不是函数的情况
  then(successCallback, failCallback) {
    successCallback =
      typeof successCallback === "function"
        ? successCallback
        : (value) => value;
    failCallback =
      typeof failCallback === "function"
        ? failCallback
        : (reason) => {
            throw reason;
          };
    let promise = new MyPromise((resolve, reject) => {
      if (this.status === FULFILLED) {
        //! 让此段代码异步以获取到then方法返回的promise实例，用于判断是否是自返回
        setTimeout(() => {
          try {
            // 状态为成功时调用成功的回调
            let returnValue = successCallback(this.value);
            MyPromise.resolvePromise(promise, returnValue, resolve, reject);
          } catch (e) {
            reject(e);
          }
        }, 0);
      } else if (this.status === REJECTED) {
        setTimeout(() => {
          try {
            // 状态为失败时调用失败的回调
            let returnValue = failCallback(this.reason);
            MyPromise.resolvePromise(promise, returnValue, resolve, reject);
          } catch (e) {
            reject(e);
          }
        }, 0);
      } else {
        // 状态未发生变化，表示处理的是异步任务
        // 保存两个回调直到状态变化时执行
        // 处理异步的情况就是将函数包裹一下，让其可以拥有错误处理的能力
        this.successCallback.push(() => {
          setTimeout(() => {
            try {
              // 状态为成功时调用成功的回调
              let returnValue = successCallback(this.value);
              MyPromise.resolvePromise(promise, returnValue, resolve, reject);
            } catch (e) {
              reject(e);
            }
          }, 0);
        });
        this.failCallback.push(() => {
          setTimeout(() => {
            try {
              // 状态为失败时调用失败的回调
              let returnValue = failCallback(this.reason);
              MyPromise.resolvePromise(promise, returnValue, resolve, reject);
            } catch (e) {
              reject(e);
            }
          }, 0);
        });
      }
    });
    return promise;
  }
  catch(failCallback) {
    return this.then(null, failCallback);
  }
  finally(callback) {
    this.then(
      (value) => MyPromise.resolve(callback()).then(() => value),
      (reason) =>
        MyPromise.resolve(callback()).then(() => {
          throw reason;
        })
    );
  }
  // 解析then回调返回的值是什么类型，用来决定then返回的promise实例的状态。
  static resolvePromise(promise2, x, resolve, reject) {
    if (promise2 === x) {
      return reject(
        new TypeError("Chaining cycle detected for promise #<myPromise>")
      );
    }

    let called = false;

    if ((typeof x === "object" && x !== null) || typeof x === "function") {
      try {
        let then = x.then;
        if (typeof then === "function") {
          // 判断是否是promise
          then.call(
            x,
            (y) => {
              // 如果 resolvePromise 以值 y 为参数被调用，则运行 [[Resolve]](promise, y)
              if (called) return;
              called = true;
              MyPromise.resolvePromise(promise2, y, resolve, reject);
            },
            (r) => {
              if (called) return;
              called = true;
              reject(r);
            }
          );
        } else {
          resolve(x);
        }
      } catch (e) {
        if (called) return;
        called = true;
        reject(e);
      }
    } else {
      // 如果 x 不为对象或者函数，以 x 普通值执行回调
      resolve(x);
    }
  }

  static all(array) {
    let result = [], // 记录结果
      index = 0; // 记录当前执行的进度
    return new MyPromise((resolve, reject) => {
      function addData(key, value) {
        result[key] = value;
        index++;
        if (index === array.length) resolve(result);
      }
      for (let i = 0; i < array.length; i++) {
        let current = array[i];
        if (current instanceof MyPromise) {
          // 是 promise 实例
          current.then(
            (value) => addData(i, value),
            (reason) => reject(reason)
          );
        } else {
          // 非 promise 实例
          addData(i, current);
        }
      }
    });
  }

  static resolve(value) {
    if (value instanceof MyPromise) return value;
    return new MyPromise((resolve) => resolve(value));
  }

  static reject(reason) {
    return new MyPromise((resolve, reject) => {
      reject(reason);
    });
  }
}

MyPromise.defer = MyPromise.deferred = function () {
  let deferred = {};

  deferred.promise = new MyPromise((resolve, reject) => {
    deferred.resolve = resolve;
    deferred.reject = reject;
  });
  return deferred;
};

// export default MyPromise;
module.exports = MyPromise;
