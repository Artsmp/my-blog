    声明两个变量：

    • 一个用于确定当前是否还处在刷新token请求的阶段
    • 另一个用于保存所有报错401的接口请求

    在 axios 响应拦截器中，判断接口是否返回 401，返回 401 的话，判断当前是否已经处在刷新 token 请求阶段了，如果还没调用过刷新 token 的请求，则调用刷新 token 的接口，当这个接口成功返回时重新设置 localStorage 中的 token 并调用失败队列中的请求并将 新token 传入。返回失败，说明 refreshToken 也失效了，则让用户重新登录。如果当前已经发送过刷新 token 的请求了，则返回一个新的 Promise，在这个 Promise 内部将该次请求包装成一个函数并存入到失败队列中，在该函数中接收一个新 token，并重新赋值给请求头中的 授权字段，然后使用新的头重新发一次请求并将结果传入到 resolve 中。

    import axios from "axios";
import { message } from "ant-design-vue";
const Base64 = require("js-base64").Base64;
import store from "../store";
import router from "../router";
import { localGet, localRemove } from "./storage";
const Message = message;
const STREAM_TYPE = "application/octet-stream";
import { reLogin } from "@/api/user/index.js";

let requestCount = 0; // 请求次数用于判断是否显示加载数据
let loadingWhiteList = ["/minio/upload"]; // 保存不需要显示加载层的接口
let isRefreshing = false; // 防止多次调用刷新token接口
let requests = []; // 当token过期时保存需要重试的接口队列

const instance = axios.create({
  baseURL: process.env.VUE_APP_BASE_API,
  timeout: 10000,
});

// 添加请求拦截器
instance.interceptors.request.use(
  function (config) {
    // 在发送请求之前做些什么
    if (requestCount === 0 && !loadingWhiteList.includes(config.url))
      store.commit("SHOW_LOADING");
    requestCount++;
    // store.commit('SHOW_LOADING')
    const token = localGet("token");
    if (token) {
      config.headers["Authorization"] =
        "Bearer " + Base64.encode(token + new Date().getTime());
    } else if (store.state.touristToken) {
      config.headers["Authorization"] =
        "Bearer " +
        Base64.encode(store.state.touristToken + new Date().getTime());
    }

    return config;
  },
  function (error) {
    // 对请求错误做些什么
    return Promise.reject(error);
  }
);

// 添加响应拦截器
instance.interceptors.response.use(
  function (response) {
    requestCount--;
    if (requestCount === 0) store.commit("HIDE_LOADING");
    // 2xx 范围内的状态码都会触发该函数。
    // 对响应数据做点什么
    const data = response.data;
    const { code, message, ret_code } = data;
    //如果是文件流直接返回
    if (response.headers["content-type"].includes(STREAM_TYPE)) {
      return data;
    } else if (code && code === 200) {
      return data.data;
    } else if (ret_code) {
      return data;
    } else if (code === 401) {
      console.log("token过期");
      // token过期
      if (!isRefreshing) {
        isRefreshing = true;
        return reLogin(localGet("refreshToken"))
          .then((res) => {
            if (res != null) {
              const { token } = res;
              store.commit("SET_TOKEN", res);
              response.headers["Authorization"] =
                "Bearer " + Base64.encode(token + new Date().getTime());
              // token 刷新后将数组的方法重新执行
              requests.forEach((cb) => cb(token));
              requests = []; // 重新请求完清空
              return instance(response.config);
            } else {
              Message.warning("登录凭证失效，请重新登录！");
              router.replace("/login");
              localRemove("token");
              localRemove("refreshToken");
            }
          })
          .finally(() => {
            isRefreshing = false;
          });
      } else {
        // 返回未执行 resolve 的 Promise
        return new Promise((resolve) => {
          // 用函数形式将 resolve 存入，等待刷新后再执行
          requests.push((token) => {
            response.headers["Authorization"] =
              "Bearer " + Base64.encode(token + new Date().getTime());
            resolve(instance(response.config));
          });
        });
      }
    }
  },
  function (error) {
    requestCount--;
    if (requestCount === 0) store.commit("HIDE_LOADING");
    // 超出 2xx 范围的状态码都会触发该函数。
    // 对响应错误做点什么
    if (error.response) {
      // 请求成功发出且服务器也响应了状态码，但状态代码超出了 2xx 的范围
      Message.error("系统服务错误，请联系管理员！");
    } else if (error.request) {
      // 请求已经成功发起，但没有收到响应
      // `error.request` 在浏览器中是 XMLHttpRequest 的实例，
      // 而在node.js中是 http.ClientRequest 的实例
      Message.error("服务未响应，请稍后重试！");
      console.log(error.request);
    } else {
      // 发送请求时出了点问题
      Message.error("请检查网络是否连接！");
      console.log("Error", error.message);
    }
    console.log(error);
    return Promise.reject(error);
  }
);

export default instance;
